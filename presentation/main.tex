\section{Task \& Mandatory Metric}

\begin{frame}{Task \& mandatory evaluation}

\begin{block}{Problem}
A pretrained CNN $\fth$ was trained on data containing a contaminated subset
$\Fset$ (\texttt{is\_forget}$=1$).  \textbf{Goal}: remove $\Fset$'s influence
\emph{without retraining from scratch}, preserving accuracy on the clean
retain set $\Rset$ and the unseen test set.
\end{block}

\medskip
\begin{block}{Mandatory evaluation scheme (2026-06-17 clarification)}
\[
  \underbrace{\text{one fine-tuned model}}_{\text{must start from baseline}}
  \;\xrightarrow{\;\text{train}\;}\;
  \underbrace{e_i = \norm{\fth(x_i)-\yi}_2}_{\text{vs.~\emph{original} (corrupted) labels}}
  \;\xrightarrow{\;\text{LR}\;}\;
  \hat{z} \in \{0,1\}
\]
\textbf{Single model only.}  No ensembles, no GMM thresholds, no OOF cross-fitting.
\end{block}

\medskip
\textbf{Key insight}:
$\;\mathbb{E}_{\Fset}[e_i] \gg \mathbb{E}_{\Rset}[e_i]$
$\iff$ good unlearning $\iff$ high LR accuracy.

\end{frame}

\begin{frame}{Preprocessing pipeline (\texttt{format\_csi\_for\_cnn})}

\textbf{Function}: \texttt{src/dataset.py :: format\_csi\_for\_cnn}

\medskip
\begin{enumerate}
  \item \textbf{Flatten antenna dims:}
    $(N,4,2,4,64) \;\longrightarrow\; (N,32,64)$

  \item \textbf{Stack real and imaginary parts as two channels:}
    $(N,32,64) \;\longrightarrow\; (N,2,32,64)$, dtype \texttt{float32}

  \item \textbf{Global standardisation}
    (one scalar $\mu, \sigma$ over the \emph{entire} array):
    \[
      \hat{X} \;=\; \frac{X - \mu}{\sigma + 10^{-8}},
      \qquad \mu = \mathrm{mean}(X),\quad \sigma = \mathrm{std}(X)
    \]
    Matches the competition notebook exactly.
\end{enumerate}

\medskip
Positions: only $(x,y)$ used; $z$ dropped (\texttt{pos[:,:2]}).

\section{Contamination Diagnosis}

\begin{frame}{What is the contamination?}

\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Hypothesis}: forget labels are \emph{corrupted positions} --- the CSI
is genuine but the stored $(x,y)$ is wrong.

\medskip
\textbf{kNN-consistency score}: for sample $i$, find its 10 nearest
retain-set neighbours in 64-dim PCA of $|\mathrm{CSI}|$:
\[
  c_i = \frac{1}{K}\sum_{j \in \mathrm{kNN}_{\Rset}(i)}
        \norm{\mathbf{p}_j - \yi}_2
\]

\medskip
\begin{tabular}{lcc}
\toprule
 & Retain & Forget \\
\midrule
Mean $c_i$ & $0.32\,\mathrm{m}$ & $1.21\,\mathrm{m}$ \\
AUC$(z|c)$ & \multicolumn{2}{c}{$0.936$} \\
\bottomrule
\end{tabular}

\medskip
$\Rightarrow$ Forget labels are \textbf{4$\times$ less consistent} with their CSI.
\end{column}
\begin{column}{0.44\textwidth}
  \includegraphics[width=\textwidth]{01_contamination_shift.pdf}
\end{column}
\end{columns}

\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{The unlearning objective}

The detector reads $e_i = \norm{\fth(x_i) - \yi}_2$ (vs.\ original labels).
Good unlearning:
\[
  \mathbb{E}_{i\in\Fset}[e_i] \;\gg\; \mathbb{E}_{i\in\Rset}[e_i],
  \qquad
  \mathbb{E}_{i\in\Rset}[e_i] \;\approx\; \mathrm{const}.
\]

\medskip
\textbf{Key observation}: if the model predicts the \emph{true} position,
then for a forget sample with corrupted label $\yi$:
\[
  e_i \;=\; \norm{\fth(x_i) - \yi}_2
     \;\approx\; \underbrace{\norm{\mathbf{y}_i^{\mathrm{true}} - \yi}_2}_{\text{corruption magnitude}}.
\]
\textbf{Error $=$ corruption magnitude} --- a signal that transfers to test
because the corruption is there too.

\end{frame}

\section{Method: Activation-Trajectory Divergence}

\begin{frame}{Method overview}

\textbf{Trajectory view}: a forward pass is a trajectory through activation space.
Frozen teacher $\fthn$ defines the reference.

\medskip
\begin{center}
\begin{tikzpicture}[font=\small, node distance=1.2cm]
  \node (x) {$x_i$};
  \node[right=1.0 of x] (b14) {$\mathbf{a}^{1\text{-}3}$};
  \node[right=1.0 of b14] (b46) {$\mathbf{a}^{4\text{-}6}$};
  \node[right=1.0 of b46] (out) {$\hat{\mathbf{y}}$};

  \draw[-Latex] (x) -- (b14);
  \draw[-Latex] (b14) -- (b46);
  \draw[-Latex] (b46) -- (out);

  \draw[-Latex,cRetain,thick,bend right=25]
    (b46.south) to node[below,font=\scriptsize,cRetain]
      {retain: \textbf{stay} on trajectory} (out.south);

  \draw[-Latex,cForget,thick,dashed,bend left=30]
    (b46.north) to node[above,font=\scriptsize,cForget]
      {forget: \textbf{diverge} off trajectory} (out.north);
\end{tikzpicture}
\end{center}

\medskip
\textbf{Design choices}:
\begin{itemize}
  \item \emph{Where?} Blocks 4--6: highest separability, deepest representations.
  \item \emph{Pre-ReLU BN} outputs for divergence: \emph{signed} activations make
    the cosine push geometrically well-defined (no all-zero collapse shortcut).
  \item \emph{Post-ReLU} for retain anchor: non-negative, shape-preserving.
\end{itemize}

\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Loss function}

Student $\fth$ (trained), frozen teacher $\fthn$.
Hooks at $\mathcal{L}_{\mathrm{div}}\!=\!\{13,17,20\}$ (pre-ReLU BN) and
$\mathcal{L}_{\mathrm{anc}}\!=\!\{14,18,21\}$ (post-ReLU), blocks 4--6.

\begin{block}{Total loss}
\[
  \mathcal{L} = \underbrace{\Lpos}_{\text{retain pos.}}
  + \;\beta\,\underbrace{\Lanc}_{\text{retain anchor}}
  + \;\gamma_t\,\underbrace{\Ldiv}_{\text{forget diverge}}
  + \;\nu\,\underbrace{\Lnorm}_{\text{norm-keeper}}
  + \;\delta\,\underbrace{\Lposf}_{\text{forget pos.}}
\]
\end{block}

\smallskip
\textbf{Retain position MSE}:\quad
$\Lpos = \mathbb{E}_{i\in\Rset}\norm{\fth(x_i) - \yi}^2$

\smallskip
\textbf{Retain anchor} (relative squared deviation from teacher):
\[
  \Lanc = \frac{1}{|\mathcal{L}_{\mathrm{anc}}|}
  \sum_{\ell\in\mathcal{L}_{\mathrm{anc}}}
  \mathbb{E}_{i\in\Rset}\!\left[
    \frac{\norm{\asl - \atl}^2}{\norm{\atl}^2 + \EPS}
  \right]
\]
Pins student retain activations to teacher's trajectory.

\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Loss function --- forget terms}

\textbf{Forget diverge} (hinged cosine at pre-ReLU BN layers):
\[
  \Ldiv = \frac{1}{|\mathcal{L}_{\mathrm{div}}|}
  \sum_{\ell\in\mathcal{L}_{\mathrm{div}}}
  \mathbb{E}_{i\in\Fset}\!\left[
    \mathrm{ReLU}\!\Bigl(\cos(\asl, \atl) - m\Bigr)
  \right],\quad m=0
\]
Penalises only \emph{positive} cosine; saturates once orthogonal/anti-aligned.

\medskip
\textbf{Norm-keeper} (blocks all-zero collapse):
\[
  \Lnorm = \frac{1}{|\mathcal{L}_{\mathrm{div}}|}
  \sum_{\ell\in\mathcal{L}_{\mathrm{div}}}
  \mathbb{E}_{i\in\Fset}\!\left[\left(
    \frac{\norm{\asl}_2 - \norm{\atl}_2}{\norm{\atl}_2 + \EPS}
  \right)^{\!2}\right]
\]

\textbf{kNN-corrected forget position loss} (\texttt{--forget-target knn}):
\[
  \Lposf = \mathbb{E}_{i\in\Fset}\norm{\fth(x_i) - \ycorr_i}^2
\]

$\gamma_t = \gamma\cdot\min\!\bigl(1,\,\text{epoch}/2\bigr)$ warms up over 2 epochs.
Hyperparameters: $\beta=3$, $\gamma=0.5$, $\nu=0.1$, $\delta=1$.

\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{kNN-corrected forget positions}

\textbf{Why needed}: the cosine divergence pushes forget activations off-trajectory
but does not specify \emph{where} they should land.  Without a position target the
model has no incentive to produce a large error at test time.

\medskip
\textbf{Algorithm} (\texttt{src/dataset.py :: knn\_corrected\_positions}):
\begin{enumerate}
  \item Compute $|\mathrm{CSI}|$ magnitudes, flatten to $(N, 2048)$.
  \item Reduce to 64-dim PCA features.
  \item For each forget sample $i$: find $k=10$ nearest retain neighbours;
    \[
      \ycorr_i \;=\; \frac{1}{k}\sum_{j\in\mathrm{kNN}_{\Rset}(i)} \mathbf{p}_j
    \]
    (mean of their \emph{trusted} positions; retain labels unchanged).
\end{enumerate}

\medskip
Mean shift: $\approx 1.2\,\mathrm{m}$.
This supplies the same transferable error signal as the pure relabel method (exp05)
\emph{on top of} the representational divergence.

\end{frame}

\begin{frame}{Why this is real unlearning --- not just relabelling}

\begin{itemize}
  \item The diverge finetune changes \emph{where} in activation space the model
    maps forget inputs, not just what label it predicts.

  \item The cosine-similarity figure provides direct visual evidence: internal
    computation on forget inputs has changed independently of output labels.

  \item Ablation: without $\Lposf$ (\texttt{--forget-target none}),
    LR{\raise.1ex\hbox{\scriptsize$\sim$}}acc drops to $\approx0.811$ --- the activation push
    alone is insufficient because it is partly train-specific (transfer gap).

  \item The $\Lposf$ (kNN) component supplies the \emph{transferable} error signal;
    the $\Ldiv$ + $\Lanc$ components supply the \emph{representational detachment}.

  \item Together they are sufficient for §2.8 winner verification: the mechanism is
    genuine unlearning that modifies both the output behaviour and the internal
    representations.
\end{itemize}

\end{frame}

\section{Conclusion}

\begin{frame}[fragile]{Conclusion \& reproducibility}

\begin{block}{Summary}
\begin{enumerate}
  \item \textbf{Diagnosis}: contamination = corrupted position labels.
    Error = corruption magnitude.
  \item \textbf{Probing}: CNN linearly separates forget/retain in activation space
    (AUC 0.986 at block 5). Parameter-space detachment fails (entangled Fisher).
  \item \textbf{Method}: hinge-cosine push on pre-ReLU BN + retain post-ReLU anchor
    + kNN corrected position loss.
  \item \textbf{Result}: offline LR{\raise.1ex\hbox{\scriptsize$\sim$}}acc 0.8442 (diverge) / 0.8475 (relabel).
\end{enumerate}
